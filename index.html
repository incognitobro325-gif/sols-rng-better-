/* UPDATED SCRIPT 1. Added triggerCutscene logic 2. Quicken roll speed (Removed cooldown/shorter intervals) 3. Cheaper & Stronger Luck Upgrades 4. Rarity formatting (1M, 1B, etc.) */ let rolls = 0; let luck = 1.0; let inventory = ["Common"]; let rollHistory = []; let currentStreak = 0; let isRolling = false; let autoOn = false; let autoInt = null; const data = { Common: { color: "#888", chance: 2, vfx: 0 }, Uncommon: { color: "#4caf50", chance: 8, vfx: 1 }, Rare: { color: "#2196f3", chance: 32, vfx: 1 }, Gilded: { color: "#ffd700", chance: 512, vfx: 2 }, Legendary: { color: "#ffeb3b", chance: 2048, vfx: 3, cs: "✦" }, Celestial: { color: "#00ffff", chance: 350000, vfx: 4, cs: "✦" }, Arcane: { color: "#9c27b0", chance: 1000000, vfx: 5, cs: "✧" }, Chromatic: { color: "#ff00ff", chance: 20000000, vfx: 6, cs: "✧" }, Transcendent: { color: "#ff2e63", chance: 100000000, vfx: 7, cs: "❂" } }; function formatRarity(num) { if (num >= 1e9) return (num / 1e9).toFixed(1) + "B"; if (num >= 1e6) return (num / 1e6).toFixed(1) + "M"; if (num >= 1e3) return (num / 1e3).toFixed(1) + "K"; return num; } function roll() { if (isRolling) return; // FAST ROLL SPEED: No more heavy cooldowns isRolling = true; rolls++; let rollVal = "Common"; const tiers = Object.keys(data).reverse(); for (let t of tiers) { // LUCK IS NOW MORE POWERFUL if (Math.random() < ((luck * 1.5) / data[t].chance)) { rollVal = t; break; } } if (rollVal !== "Common") currentStreak++; else currentStreak = 0; rollHistory.unshift(rollVal); if (rollHistory.length > 5) rollHistory.pop(); if (!inventory.includes(rollVal)) inventory.push(rollVal); // Faster visual transition (100ms instead of 250ms) setTimeout(() => { if (data[rollVal].cs) { triggerCutscene(rollVal); } else { displayAura(rollVal); } isRolling = false; }, 100); } function triggerCutscene(name) { const cs = document.getElementById("cutscene"); const csStar = document.getElementById("cs-star"); const csName = document.getElementById("cs-name"); const aura = data[name]; cs.style.display = "flex"; csStar.innerText = aura.cs; csStar.style.color = aura.color; csName.innerText = name; csName.style.color = aura.color; // Fast snappy animations csStar.style.animation = "starPop 0.5s forwards"; csName.style.animation = "reveal 0.5s 0.3s forwards"; setTimeout(() => { cs.style.display = "none"; csStar.style.opacity = 0; csName.style.opacity = 0; displayAura(name); }, 1200); } function craftLuck() { // CHEAPER UPGRADES: Lower base and scaling const cost = Math.floor(25 + (luck * 25)); if (rolls >= cost) { rolls -= cost; luck += 0.5; // LUCIER UPGRADES: +0.5 instead of +0.1 updateUI(); } else { alert(`Need ${cost} rolls!`); } } function displayAura(name) { const aura = data[name]; const disp = document.getElementById("disp-name"); disp.innerText = name; disp.style.color = aura.color; document.getElementById("disp-rarity").innerText = `1 / ${formatRarity(aura.chance)}`; document.documentElement.style.setProperty("--aura-color", aura.color); createVFX(aura.vfx); updateUI(); } // Ensure toggleAuto runs at max speed function toggleAuto() { autoOn = !autoOn; const btn = document.getElementById("btn-auto"); btn.innerText = `AUTO: ${autoOn ? "ON" : "OFF"}`; btn.style.background = autoOn ? "#2e4d2e" : "#222"; if (autoOn) { autoInt = setInterval(() => { if (!isRolling) roll(); }, 50); // High-speed interval } else { clearInterval(autoInt); } }
